cmake_minimum_required(VERSION 3.20)
project(app)

set(CMAKE_CXX_STANDARD 20)

set(LIB_DIR ${CMAKE_SOURCE_DIR}/lib)
set(INCLUDE_DIR ${CMAKE_SOURCE_DIR}/include)

find_package(glfw3 REQUIRED)
find_package(OpenGL REQUIRED)

add_library(glad STATIC ${LIB_DIR}/glad/src/glad.c)
target_include_directories(glad PUBLIC ${LIB_DIR}/glad/include)

add_executable(${PROJECT_NAME}
    main/main.cpp
    main/core/window/Window.cpp
    main/core/renderer/Program.cpp
    main/core/renderer/mesh/Mesh.cpp
    main/core/renderer/Renderer.cpp
    main/core/renderer/mesh/objects/VBO.cpp
    main/core/renderer/mesh/objects/VAO.cpp
    main/core/renderer/mesh/objects/EBO.cpp

    main/utils/Parse.cpp
)
target_include_directories(${PROJECT_NAME} PUBLIC ${INCLUDE_DIR})


target_link_libraries(${PROJECT_NAME}
    PRIVATE
    OpenGL::GL
    glfw
    glad
)
